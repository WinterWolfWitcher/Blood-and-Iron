    function saveCharacter() {
        const fields = [
            'name', 'titles', 'age', 'origin', 'residence', 'standing', 'occupation',
            'ancestry', 'marks', 'family', 'divine', 'height', 'build', 'coloring',
            'features', 'bearing', 'voice', 'condition', 'primary', 'secondary',
            'armor', 'possessions', 'companion', 'temperament', 'virtues', 'vices',
            'passions', 'fears', 'faith', 'martial', 'knowledge', 'practical',
            'supernatural', 'liege', 'familyties', 'companions', 'romance', 'enemies',
            'debts', 'publicrep', 'noble', 'church', 'deeds', 'earlylife', 'defining',
            'current', 'secrets', 'goals', 'hope', 'darkfear', 'prophecy',
            'appearance', 'notes', 'player', 'contact', 'created', 'updated'
        ];

        const character = {};
        fields.forEach(field => {
            character[field] = document.getElementById(field).value;
        });

        const dataStr = JSON.stringify(character, null, 2);
        const dataBlob = new Blob([dataStr], { type: 'application/json' });
        const url = URL.createObjectURL(dataBlob);
        const link = document.createElement('a');
        link.href = url;
        link.download = `${character.name || 'character'}_chronicle.json`;
        link.click();
        URL.revokeObjectURL(url);
    }

    function loadCharacter() {
        const input = document.createElement('input');
        input.type = 'file';
        input.accept = '.json';
        input.onchange = (e) => {
            const file = e.target.files[0];
            const reader = new FileReader();
            reader.onload = (event) => {
                const character = JSON.parse(event.target.result);
                Object.keys(character).forEach(field => {
                    const element = document.getElementById(field);
                    if (element) {
                        element.value = character[field];
                    }
                });
            };
            reader.readAsText(file);
        };
        input.click();
    }

    function clearForm() {
        if (confirm('Are you certain you wish to clear all fields? This cannot be undone.')) {
            document.querySelectorAll('input, textarea, select').forEach(element => {
                element.value = '';
            });
        }
    }

    const today = new Date().toLocaleDateString('en-GB', { 
        day: 'numeric', 
        month: 'long', 
        year: 'numeric' 
    });
    document.getElementById('created').value = today;
    document.getElementById('updated').value = today;
</script>
